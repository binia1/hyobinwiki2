        async function goToLink(filename) {
            // 1. 일반적인 나무위키 키워드인 경우 (예외 처리)
            const namuKeywords = ['대한민국', '국립대학', '사립대학'];
            const pureName = filename.replace('.html', '').replace(/_/g, ' ');
            
            if (namuKeywords.includes(pureName) || filename.includes('namu.wiki')) {
                window.open(`https://namu.wiki/w/${encodeURIComponent(pureName)}`, '_blank');
                return;
            }

            // 2. 로컬 파일 이동
            try {
                location.href = filename;
            } catch (e) {
                showMsg('문서를 찾을 수 없습니다.');
            }
        }
        
        // 목차 스무스 스크롤
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                const targetElement = document.querySelector(targetId);
                
                if(targetElement){
                    window.scrollTo({
                        top: targetElement.offsetTop - 80, 
                        behavior: 'smooth'
                    });
                }
            });
        });
